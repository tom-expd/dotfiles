** encrypt

cd dotfile.secret

** decrypt

#+BEGIN_SRC bash
mkdir -p ssh/.ssh
gpg -d dotfile.secret/ssh-config.gpg > ssh/.ssh/config

gpg -d dotfile.secret/emacs-amazon-config.org.gpg > emacs/.emacs.d/amazon-config.org

mkdir -p pg
gpg -d dotfile.secret/pgpass.gpg > pg/.pgpass
chmod 0600 ~/.pgpass
#+END_SRC
